JOB TYPES TABLE CREATE TABLE job_types ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), name TEXT NOT NULL UNIQUE, color TEXT DEFAULT '#3B82F6', is_active BOOLEAN DEFAULT true, sort_order INT DEFAULT 0 ); -- Insert default job types INSERT INTO job_types (name, color, sort_order) VALUES ('Water Damage', '#3B82F6', 1), ('Fire Restoration', '#EF4444', 2), ('Mold Remediation', '#10B981', 3), ('Inspection', '#8B5CF6', 4); -- 2. RECURRING JOBS TABLE CREATE TABLE recurring_jobs ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW(), title TEXT NOT NULL, job_type_id UUID REFERENCES job_types(id), job_type_name TEXT, team_id UUID REFERENCES teams(id), team_name TEXT, frequency TEXT DEFAULT 'weekly', -- daily, weekly, biweekly, monthly day_of_week TEXT DEFAULT 'Monday', time_of_day TIME DEFAULT '08:00', location TEXT, is_active BOOLEAN DEFAULT true ); -- Trigger for recurring_jobs updated_at CREATE TRIGGER recurring_jobs_updated_at BEFORE UPDATE ON recurring_jobs FOR EACH ROW EXECUTE FUNCTION update_updated_at(); -- 3. RLS POLICIES ALTER TABLE job_types ENABLE ROW LEVEL SECURITY; ALTER TABLE recurring_jobs ENABLE ROW LEVEL SECURITY; CREATE POLICY "Anyone can read job_types" ON job_types FOR SELECT TO anon, authenticated USING (true); CREATE POLICY "Anyone can read recurring_jobs" ON recurring_jobs FOR SELECT TO anon, authenticated USING (true); -- Insert sample recurring job INSERT INTO recurring_jobs (title, job_type_name, team_name, frequency, day_of_week, time_of_day, location) VALUES ('Equipment Inspection', 'Inspection', 'Team A', 'weekly', 'Monday', '08:00', 'Main Office'); -- ============================================= -- DONE!
